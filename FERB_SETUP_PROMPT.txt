TAREFA 5: CRIAR ARQUIVO backend/src/index.js

Conteúdo:
// FERB - Backend Server
// Agente que me cobra todo dia

require('dotenv').config();
const express = require('express');
const cors = require('cors');

const app = express();
const PORT = process.env.PORT || 3000;

// Middleware
app.use(express.json());
app.use(cors());

// Health check - verifica se servidor está rodando
app.get('/health', (req, res) => {
  res.json({ 
    status: 'FERB está vivo! 🤖',
    timestamp: new Date()
  });
});

// Placeholder - dados do Chartmetric virão aqui
app.get('/api/artist', (req, res) => {
  res.json({
    message: 'Dados do Chartmetric',
    status: 'Conectando na próxima tarefa',
    streams: 0,
    listeners: 0
  });
});

// Iniciar servidor
app.listen(PORT, () => {
  console.log(`🚀 FERB backend rodando em http://localhost:${PORT}`);
  console.log(`Teste: curl http://localhost:${PORT}/health`);
});
